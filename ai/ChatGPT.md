procedure GrowLand(Chance: Single);
var
  x, y, dx, dy, n: Integer;
  Temp: array of array of Boolean;
begin
  Temp := Land;

  for y := 1 to H-2 do
    for x := 1 to W-2 do
      if not Land[y][x] then
      begin
        n := 0;
        for dy := -1 to 1 do
          for dx := -1 to 1 do
            if Land[y+dy][x+dx] then Inc(n);

        if (n > 0) and (Random < Chance) then
          Temp[y][x] := True;
      end;

  Land := Temp;
end;
